010package org.kew.rmf.transformers.authors;
011
012import org.apache.commons.lang3.StringUtils;
013import org.kew.rmf.transformers.SafeStripNonAlphasTransformer;
014import org.kew.rmf.transformers.StringShrinker;
015import org.kew.rmf.transformers.Transformer;
016
017/**
018 * This transformer tries to identify all publication authors of plant names in
019 * a string and returns a string where each of their surnames are shrunk/cropped
020 * to a length of `shrinkTo`.
021 *
022 * For examples of standard usage and corner cases see {@link
023 * ShrunkPubAuthorsTest}
024 */
025public class ShrunkPubAuthors implements Transformer {
026
027    private Integer shrinkTo = null;
028
029    @Override
030    public String transform(String s) {
031        s = new DotFDotCleaner().transform(s);
032        s = new CleanedPubAuthors().transform(s);
033        s = new SurnameExtractor().transform(s);
034        s = new SafeStripNonAlphasTransformer().transform(s);
035        s = s.replaceAll("\\s+", " ");
036        // shrink each identified author surname to shrinkTo if set
037        if (this.shrinkTo != null) {
038            String[] toShrink = s.split(" ");
039            for (int i=0;i<toShrink.length;i++) {
040                toShrink[i] = new StringShrinker(this.shrinkTo).transform(toShrink[i]);
041            }
042            s = StringUtils.join(toShrink, " ");
043        }
044        return s.toLowerCase();
045    }
046
047    public Integer getShrinkTo() {
048        return shrinkTo;
049    }
050
051    public void setShrinkTo(Integer shrinkTo) {
052        this.shrinkTo = shrinkTo;
053    }
054
055}
