010package org.kew.rmf.transformers.authors;
011
012import java.util.ArrayList;
013import java.util.List;
014
015import org.apache.commons.lang3.StringUtils;
016import org.kew.rmf.transformers.RegexDefCollection;
017import org.kew.rmf.transformers.SafeStripNonAlphanumericsTransformer;
018import org.kew.rmf.transformers.StringShrinker;
019import org.kew.rmf.transformers.Transformer;
020
021/**
022 * This transformer tries to identify *all* authors (accepts publishing-,
023 * basionym-, ex-, in-) of plant names in a string and returns a string where
024 * each of their surnames are shrunk/cropped to a length
025 * of `shrinkTo`.
026 *
027 * For examples of standard usage and corner cases see {@link ShrunkAuthorsTest}
028 */
029public class ShrunkAuthors extends RegexDefCollection implements Transformer {
030
031    private Integer shrinkTo = null;
032
033    @Override
034    public String transform(String s) {
035        s = new DotFDotCleaner().transform(s);
036        s = new SurnameExtractor().transform(s);
037        String pub = new StripBasionymAuthorTransformer().transform(s);
038        String bas = new StripPublishingAuthorTransformer().transform(s);
039        String exIn = String.format("%s|%s", EX_MARKER_REGEX, IN_MARKER_REGEX);
040        List<String> surnames = new ArrayList<>();
041        for (String authors: new String[] {bas, pub}) {
042            authors = authors.replaceAll(exIn, " ");
043            authors = authors.replaceAll("\\s+", " ");
044            authors = new SafeStripNonAlphanumericsTransformer().transform(authors);
045            for (String author:authors.split(" ")) {
046                // shrink each identified author surname to shrinkTo if set
047                if (this.shrinkTo != null) author = new StringShrinker(this.shrinkTo).transform(author);
048                if (!StringUtils.isBlank(author)) surnames.add(author.trim());
049            }
050        }
051        return StringUtils.join(surnames, " ").toLowerCase();
052    }
053
054    public Integer getShrinkTo() {
055        return shrinkTo;
056    }
057
058    public void setShrinkTo(Integer shrinkTo) {
059        this.shrinkTo = shrinkTo;
060    }
061
062}
